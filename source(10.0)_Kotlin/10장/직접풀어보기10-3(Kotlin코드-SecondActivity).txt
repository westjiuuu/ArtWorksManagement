package com.cookandroid.self13_3


import android.app.Activity
import androidx.appcompat.app.AppCompatActivity
import android.content.Intent
import android.os.Bundle
import android.view.View
import android.widget.Button

class SecondActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.second)
        title = "Second 액티비티"

        val inIntent = intent

        val calc = inIntent.getStringExtra("Calc")

        var calValue = 0
        if (calc == "+") {
            calValue = inIntent.getIntExtra("Num1", 0) + inIntent.getIntExtra("Num2", 0)
        } else if (calc == "-") {
            calValue = inIntent.getIntExtra("Num1", 0) - inIntent.getIntExtra("Num2", 0)
        } else if (calc == "*") {
            calValue = inIntent.getIntExtra("Num1", 0) * inIntent.getIntExtra("Num2", 0)
        } else {
            calValue = inIntent.getIntExtra("Num1", 0) / inIntent.getIntExtra("Num2", 0)
        }

        val retValue = calValue

        val btnReturn = findViewById<View>(R.id.btnReturn) as Button
        btnReturn.setOnClickListener {
            val outIntent = Intent(applicationContext,
                MainActivity::class.java)
            outIntent.putExtra("Hap", retValue)
            setResult(Activity.RESULT_OK, outIntent)
            finish()
        }
    }

}
